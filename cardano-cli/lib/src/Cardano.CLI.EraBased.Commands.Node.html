renderNodeCmds :: forall era. NodeCmds era -> Text
renderNodeCmds = \case
  NodeKeyGenColdCmd   {} -> Text
"node key-gen"
  NodeKeyGenKESCmd    {} -> Text
"node key-gen-KES"
  NodeKeyGenVRFCmd    {} -> Text
"node key-gen-VRF"
  NodeKeyHashVRFCmd   {} -> Text
"node key-hash-VRF"
  NodeNewCounterCmd   {} -> Text
"node new-counter"
  NodeIssueOpCertCmd  {} -> Text
"node issue-op-cert"
